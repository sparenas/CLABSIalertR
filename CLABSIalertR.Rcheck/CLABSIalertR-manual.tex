\HeaderA{add\_spc\_rules}{Add SPC rule flags (Western Electric style) to a CLABSI dataset}{add.Rul.spc.Rul.rules}
%
\begin{Description}
Add SPC rule flags (Western Electric style) to a CLABSI dataset
\end{Description}
%
\begin{Usage}
\begin{verbatim}
add_spc_rules(df, z_col = "z", y_col = "clabsi")
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{df}] A data frame that already contains at least:
\begin{itemize}

\item{} a numeric column \code{z} (z-scores)
\item{} a numeric column \code{clabsi} (counts)

\end{itemize}


\item[\code{z\_col}] Name of the z-score column (default "z")

\item[\code{y\_col}] Name of the response/measurement column (default "clabsi")
\end{ldescription}
\end{Arguments}
%
\begin{Value}
The same data frame with logical columns:
rule\_point\_3sigma, rule\_run\_8\_side, rule\_trend\_6, spc\_any
\end{Value}

\HeaderA{check\_dispersion}{Check dispersion of CLABSI counts}{check.Rul.dispersion}
%
\begin{Description}
Calculates the variance-to-mean ratio for CLABSI counts to assess
whether the data follow a Poisson distribution.
Ratios near 1 suggest Poisson behavior;
smaller than 1 indicates underdispersion (stable process);
bigger than 1 suggests overdispersion (extra variability).
\end{Description}
%
\begin{Usage}
\begin{verbatim}
check_dispersion(data)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{data}] Data frame with a numeric column \code{clabsi}
\end{ldescription}
\end{Arguments}
%
\begin{Value}
Numeric value of the variance-to-mean ratio
\end{Value}
%
\begin{Examples}
\begin{ExampleCode}
df <- data.frame(clabsi = c(1,0,0,2,1,3,0,0,1,0,2,4))
check_dispersion(df)
\end{ExampleCode}
\end{Examples}
